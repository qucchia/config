* Emacs

#+begin_src emacs-lisp :noweb yes :tangle ../home/.emacs.d/qucchia-theme.el

  (require 'autothemer)

  (autothemer-deftheme
    qucchia "qucchia theme"

    ((((class color) (min-colors #xFFFFFF))) ;; Use graphical Emacs
       <<emacs-colours>>)

    (
    <<emacs-faces>>))

  (provide-theme 'qucchia)

#+end_src
